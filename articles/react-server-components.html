📊 Code Examples

typescript
// React Server Components Explained — Core Implementation Pattern
interface Config {
  enabled: boolean;
  timeout: number;
  retries: number;
}

class Implementation {
  private config: Config;
  constructor(config: Partial = {}) {
    this.config = { enabled: true, timeout: 5000, retries: 3, ...config };
  }
  async execute(input: string): Promise {
    if (!this.config.enabled) throw new Error('Disabled');
    return this.processWithRetry(input);
  }
  private async processWithRetry(input: string, attempt = 1): Promise {
    try {
      return await this.process(input);
    } catch (err) {
      if (attempt >= this.config.retries) throw err;
      await new Promise(r => setTimeout(r, 1000 * attempt));
      return this.processWithRetry(input, attempt + 1);
    }
  }
  private async process(input: string): Promise {
    return `Processed: ${input}`;
  }
}

const impl = new Implementation({ timeout: 3000 });
const result = await impl.execute('input data');
console.log(result);
typescript
// React Server Components Explained — Testing Pattern
import { describe, it, expect, vi } from 'vitest';

describe('Implementation', () => {
  it('processes valid input', async () => {
    const impl = new Implementation();
    const result = await impl.execute('test input');
    expect(result).toContain('test input');
  });

  it('retries on failure', async () => {
    const impl = new Implementation({ retries: 3 });
    const processSpy = vi.spyOn(impl as any, 'process')
      .mockRejectedValueOnce(new Error('Temporary failure'))
      .mockResolvedValueOnce('Success after retry');
    const result = await impl.execute('input');
    expect(result).toBe('Success after retry');
    expect(processSpy).toHaveBeenCalledTimes(2);
  });

  it('throws when disabled', async () => {
    const impl = new Implementation({ enabled: false });
    await expect(impl.execute('test')).rejects.toThrow('Disabled');
  });
});

⚠️ Common Mistakes

⚠️ Premature Optimization
Don't optimize before you have metrics showing a problem. Build correctly first, measure, then optimize the actual bottleneck.
⚠️ Missing Observability
Without logging, metrics, and tracing, you're flying blind in production. Instrument your code from the start.

✅ Best Practices

  • ✅ Start with the simplest implementation that works, then iterate
  • ✅ Write tests before writing implementation code (TDD)
  • ✅ Document architecture decisions and the reasoning behind them
  • ✅ Monitor key metrics from day one and set alerts for anomalies
